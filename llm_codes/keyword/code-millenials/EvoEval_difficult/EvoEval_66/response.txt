
    ascii_sum_s = 0
    ascii_sum_t = 0
    for char in s:
        if char.isupper():
            ascii_sum_s += ord(char)
    for char in t:
        if char.islower():
            ascii_sum_t += ord(char)
    return (ascii_sum_s, ascii_sum_t)
